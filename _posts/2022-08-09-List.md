## sort = 순서대로 나열 하기
num_list = [7,4,3,6,1,2]
mix_list = ["강냉이", "강아지"]
num_list.sort()
print(num_list)

## reverse = 순서 반대로 나열하기
num_list.reverse()
print(num_list)

## clear = 모두 지우기
num_list.clear()
print(num_list)

## extend = 리스트 확장 합치기
num_list.extend(mix_list)
print(num_list)
